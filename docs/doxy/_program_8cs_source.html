101  private static bool ParseArgs(string[] args)
102  {
103  for (int i = 0; i < args.Length; i++)
104  {
105  string arg = args[i].ToLower();
106 
107  if (!args.Contains("/g") && _startGui) // Only allocate console when not using gui(/g).

109 
110  if (!AllowedArgs.Contains(arg))

112  InvalidSyntax();
113  return false;
114  }
115 
116  if (arg.StartsWith("/h") || arg.StartsWith("/?") || arg.StartsWith("--help"))
117  {
118  PrintHelp();
119  return false;
120  }
121 
122  if (arg.StartsWith("/b"))
123  {
124  if (!ParseBusinessLogicTypeArg(arg))
125  return false;
126  }
127 
128  if (arg.StartsWith("/d"))
129  {
130  if (!ParseDatabaseTypeArg(arg))
131  return false;
132  }
133 
134  if (arg.StartsWith("/s"))
135  {
136  if (!ParseSortType(arg))
137  return false;
138  }
139  }
140 
141  return true;
142  }
