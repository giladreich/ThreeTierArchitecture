134  public void DeleteTeam()

138  Team team = _view.TeamsListBox[_view.TeamSelectedIndex].ToTeam();
139  if (team == null) return;

143 
144  // Change all players of the deleted team to Unsigned Team("0").
145  if (teamPlayers.Count > 0)
146  for (int i = teamPlayers.Count - 1; i >= 0; i--)

149  Strategy.RemoveTeam(team.Id);
